# Event: initialized

HTML: 
< div class="ali-owl-btn">

JS

                    var owl = $('.owl-carousel');
                    owl.on('initialized.owl.carousel', function(event){ 
                        var _count_item = event.item.count;
                        var _elWrap = $('.ali-owl-btn');
                        for(var i=0; i<_count_item; i++) {
                            if(i == 0)
                                $(_elWrap).append("< button class=\"btn btn-default btn-hvr-pink active\" owl-index=\""+i+"\">"+(i+1)+"");
                            else 
                                $(_elWrap).append("< button class=\"btn btn-default btn-hvr-pink\" owl-index=\""+i+"\">"+(i+1)+"");
                        }
                    });

# Event: changed

JS

owl.on('changed.owl.carousel', function(event) {
                    var _owl_index = event.page.index;
                    var _buttons = $('.ali-owl-btn').find('button');
                    var _data = _json_data[_owl_index];

                    $('.ali-owl--title').text(_data.title);
                    $('.ali-owl--sub-title').text(_data.sub_title);
                    $('.ali-owl--description').text(_data.description);

                    _buttons.each(function(index, el) {
                        $(el).removeClass('active');
                        $(_buttons[_owl_index]).addClass('active');
                    });
                })
